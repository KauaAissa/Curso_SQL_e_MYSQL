CREATE PROCEDURE employeesWithHighSalary(
	IN highSalary INT
)
BEGIN
	SELECET ep.first_name, sa.salary FROM employees AS ep
	JOIN salaries AS sa ON ep.emp_no = sa.emp_no
	WHERE sa.salary > highSalary;
END//

CALL employeesWithHighSalary(80000);
